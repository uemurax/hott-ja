\begin{myBlock}{000F}{myNotation}
  \begin{enumerate}
  \item \myInlineMath{\myUniverseEl{\myPhantom{A}}}はよく省略する。
    つまり、要素\myInlineMath{A \myElemOf \myUniverse{i}}
    そのものを型とみなす。
  \item \myInlineMath{\myUniverseLift{\myPhantom{A}}}はよく省略する。
    つまり、要素\myInlineMath{A \myElemOf \myUniverse{i}}
    は\myInlineMath{\myUniverse{\myLevelSucc{i}}}の要素でもあるとみなす。
  \item \myInlineMath{\myUniverse{i}}と
    \myInlineMath{\myUniverseName{i}}を表記上区別しない。
    つまり、\myInlineMath{\myUniverse{i}}そのものを
    \myInlineMath{\myUniverse{\myLevelSucc{i}}}の要素とみなす。
  \end{enumerate}
\end{myBlock}
